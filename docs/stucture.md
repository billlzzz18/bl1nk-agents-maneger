export class AgentOrchestrator {

    activeAgents: [],
    completedTasks: [],
    blockers: []

  // à¸¥à¸‡à¸—à¸°à¹€à¸šà¸µà¸¢à¸™ Agent

  // à¸£à¸±à¸™ Task à¹€à¸”à¸µà¹ˆà¸¢à¸§

    if (!agent) {
      this.workflowState.blockers.push(
        `âŒ Agent not found: ${task.assignedAgent}`
      );
      return;
    }

    try {
      this.workflowState.activeAgents.push(agent.name);
      task.status = 'in-progress';

        data: { task },
        context: this.workflowState

      } else {

      }
    } catch (error) {

      this.workflowState.blockers.push(`Error: ${error}`);
    } finally {

        this.workflowState.activeAgents.filter(a => a !== agent.name);
    }

        success: false,
        error: String(error),
        timestamp: new Date().toISOString(),

      {
        id: `${task.id}-2`,
        title: `Implement: ${task.title}`,
        type: 'implementation',
        owner: 'Reviewer'
      },
      {
        id: `${task.id}-3`,
        title: `Document: ${task.title}`,
